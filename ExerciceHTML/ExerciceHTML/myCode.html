void RemplacerTout(string & content)
{
     for (int i = 0; i != content.size(); ++i)
     {
          if (content[i] == ' < ')
          {
               content.replace(i, 1, " & lt ");
               i += 4;
          }
          else if (content[i] == ' > ')
          {
               content.replace(i, 1, " & gt ");
               i += 4;
          }
          else if (content[i] == ' & ')
          {
               content.replace(i, 1, " & amp ");
               i += 4;
          }
          else if (content[i] == '\n')
          {
               content.replace(i, 1, " < /br > ");
               i += 6;
          }
          else if (content[i] == '\t')
          {
               content.replace(i, 1, " & nbsp & nbsp & nbsp & nbsp ");
               i += 21;
          }
     }
    
}
